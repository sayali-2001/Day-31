create database AddressBook;
use AddressBook;
create table Address_book(id int unsigned NOT NULL AUTO_INCREMENT, 
first_name varchar(15), last_name varchar(10) , city varchar(50),
state varchar(50), 
zip int, 
phoneNo long, 
email varchar(100), 
primary key(id)
);
